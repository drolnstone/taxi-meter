function getEaster(y){ const a=y%19,b=Math.floor(y/100),c=y%100,d=Math.floor(b/4),e=b%4 const f=Math.floor((b+8)/25),g=Math.floor((b-f+1)/3) const h=(19*a+b-d-g+15)%30,i=Math.floor(c/4),k=c%4 const l=(32+2*e+2*i-h-k)%7,m=Math.floor((a+11*h+22*l)/451) return new Date(y, Math.floor((h+l-7*m+114)/31)-1, ((h+l-7*m+114)%31)+1) } function sameDay(a,b){ return a.getFullYear()===b.getFullYear() && a.getMonth()===b.getMonth() && a.getDate()===b.getDate() } function lastMonday(y,m){ let d=new Date(y,m+1,0) while(d.getDay()!==1) d.setDate(d.getDate()-1) return d } function firstMonday(y,m){ let d=new Date(y,m,1) while(d.getDay()!==1) d.setDate(d.getDate()+1) return d } function substituted(y,m,day){ let d=new Date(y,m,day) if(d.getDay()===0) d.setDate(d.getDate()+1) else if(d.getDay()===6) d.setDate(d.getDate()+2) return d } function isUKBankHoliday(now){ const y = now.getFullYear() const easter = getEaster(y) const goodFriday = new Date(easter); goodFriday.setDate(easter.getDate()-2) const easterMonday = new Date(easter); easterMonday.setDate(easter.getDate()+1) const bankHolidays = [ substituted(y,0,1), // New Year's Day goodFriday, easterMonday, firstMonday(y,4), // Early May lastMonday(y,4), // Spring lastMonday(y,7), // Summer substituted(y,11,25), // Christmas substituted(y,11,26) // Boxing Day ] return bankHolidays.some(bh => sameDay(now, bh)) } function isExtraRate(now){ const y = now.getFullYear() const ranges = [ [new Date(y,11,24,18,0,0), new Date(y,11,27,6,0,0)], // Christmas Eve 6pm → Dec 27 6am [new Date(y,11,31,18,0,0), new Date(y+1,0,2,6,0,0)], // NYE 6pm → Jan 2 6am [new Date(y-1,11,31,18,0,0), new Date(y,0,2,6,0,0)] // catch Jan 1–2 ] return ranges.some(([start,end]) => now >= start && now < end) } function updateTariff(){ const now = new Date() const hour = now.getHours() if(isExtraRate(now)){ activeTariff = TARIFFS.Extra currentTariff = "Extra" tariffElem.innerText = "Extra" tariffElem.style.background = "#4a1a1a" tariffElem.style.color = "#ff6b6b" } else if(isUKBankHoliday(now) || hour < 6 || hour >= 21){ activeTariff = TARIFFS.Night currentTariff = "Night" tariffElem.innerText = "Night" tariffElem.style.background = "#2c1f4a" tariffElem.style.color = "#bf9fff" } else { activeTariff = TARIFFS.Day currentTariff = "Day" tariffElem.innerText = "Day" tariffElem.style.background = "#333" tariffElem.style.color = "#fff" } }
